# 通用环境

# 提升 webpack 和 Node 版本

# 将 loader 应用于最少数量的必要模块

{
  test: /\.js$/,
  include: path.resolve(__dirname,'src'),
  use: {
    loader: ['babel-loader', 'eslint-loader'],
  }

# 每个额外的 loader、plugin 都有其启动时间，尽量少的使用工具

# 解析

1、减少 resolve.modules、resolve.extentions、resolve.mainFiles、resolve.descriptionFiles 中条目数量，以免增加文件系统调用的次数

2、不适用 symlinks(npm link)，可以设置 resolve.symlinks:false

3、使用自定义 resolve plugin 规则时，没有指定上下文，可以设置 resolve.cacheWithContext:false

# 体积

1、使用数量更少、体积更小的 lib

2、多页面时，使用 SplitChunksPlugin，开启 async 模式

3、移除未引用的代码

4、只编译当前正在开发的代码

# 持久化缓存

1、在配置中使用 cache 选项

2、使用 package.json 中的 postinstall 清除缓存目录

3、将 cache 类型设置为内存或文件系统，type=memory

# SourceMap

说明：源代码到构建代码的映射技术

实现

devtool: 'cheap-module-source-map' //外部文件
//错误代码的准确信息,源代码的错误位置
/*
      错误代码的准确信息,源代码的错误位置
    inline-source-map:内联，只生成一个，构建速度更快
      错误代码的错误原因，没有错误位置
    hidden-source-map:外部
      错误代码的准确信息,源代码的错误位置
    eval-source-map:内联，生成多个，在eval函数中
      每个module会封装到eval里包裹起来执行，并且在末尾追加注释//@sourceURL
    eval(默认)
      生成一个SourceMap文件
    cheap-source-map
      生成一个没有列信息的sourcemap文件，不包含loader的sourcemap
    cheap-module-source-map
      生成一个没有列信息的sourcemap文件，loader的sourcemap被简化为只包含对应行的
  */

# 缓存

说明：第二次构建，会读取之前的缓存

1、babel 缓存

//babel缓存.js文件
options: {
  cacheDirectory: true
}

2、文件缓存

① 利用 hash 值

原理：每一次 webpack 构建时都会生成唯一的 hash 值

问题：js 和 css 使用同一个 hash 值，在只改动一个文件时，缓存失效

② 利用 chunkhash

原理：根据 chunk 生成的 hash 值，如果打包来源于同一个 chunk，那么 hash 值就一样

问题：js 和 css 的 hash 值还是一样的，css 是在 js 中被引入的，同属于一个 chunk

③ 利用 contenthash

原理：根据文件的内容生成 hash 值

output: {
  filename: 'bundle.[contenthash:10].js'
}

new MinCssExtractPlugin({
  filename: 'css/bundle.[contenthash:10].css'
})

3、第三方依赖缓存

splitChunks: {
      cacheGroup: {
        vendor: {
          test: /[\\/]node_modules[\\/]/,
          name: 'vendor',
          chunks: 'all'
        }
      }
    }

# code split

说明：代码分割、按需加载

① 多入口打包，多页面

//自动分析多入口chunk中，有没有公共文件，有的话会单独打包成一个单独的chunk
module.exports = {
  entry: {
    main: './src/js/index.js',
    test: './src/js/test.js'
  },
  optimization: {
    splitChunks: {
      chunks: 'all'
    }
  }
}

② 单入口

//将node_modules代码单独打包一个chunk输出
module.exports = {
  entry: './src/index.js',
  optimization: {
    splitChunks: {
      chunks: 'all'
    }
  }
}

③ 使用 JS 代码实现某个文件的单独打包

import(/*webpackChunkName:'test'*/ './test.js')
  .then(res => {
    console.log(res)
  })
  .catch(err => err)

# 懒加载

import(/*webpackChunkName:'test',webpackPrefetch:true*/ './test').then(res => {
  console.log(res)

plugins: [

    //快速启动，去除旧的servicework，生成servicework配置文件
    clientsClaim: true,
    skipWaiting: true
  })

//main.js
//注册servicework
if ('serviceWorker' in navigator) {
  window.addEventListener('load', function () {
    navigator.serviceworker
      .register('./service-work.js')
      .then(() => {
        console.log('注册成功')
      })
      .catch(err => err)
  })
}

# dll

说明：对第三方库进行单独打包，为更改不频繁的代码生成单独的编译结果

//webpack.dll.js
//webpack --config webpack.dll.js
module.exports = {
  entry:{
    jquery:['JQuery']
  },
  output:{
    filename:'[name].js',
    path:resolve(__dirname,'dll')
    library:'[name]_[hash]'   //定义打包的库暴露出的内容的名字
  },
  plugins:[
    new webpack.DllPlugin({
      name:'[name]_[hash]',
      path:resolve(__diranme,'dll/mainfest.json')
    })
    //webpack.config.js
    //告知webpack哪些库不参与打包
    //add-asset-html-webpack-plugin
    new webpack.DllReferencePlugin({
      mainfest:resolve(__diranme,'dll/mainfest.json')
    }),
    //自动引入
    new AddAsentHtmlWebpackPlugin({
      filepath:resolve(__dirname,'dll/jquery.js')
    })
  ]
}
